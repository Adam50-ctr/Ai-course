🤖 Building the AI Model

🎯 Goal: Create the neural network architecture
⏱️ Time: 20-25 minutes
🧠 Result: Complete CNN model ready for training

🏗️ Understanding Our CNN Architecture

We'll build a Convolutional Neural Network (CNN) - perfect for images:

🏗️ Our Model Architecture:

Input (150x150x3) → Conv2D(32) → MaxPool →
Conv2D(64) → MaxPool →
Flatten → Dense(64) → Dense(2) → Output

💻 Step 1: Create the Training Script

Create our main training file: train_model.py

import tensorflow as tf
from tensorflow.keras.preprocessing.image import ImageDataGenerator
import matplotlib.pyplot as plt
import os

print("🤖 Building Cat vs Dog AI Model")
print("=" * 40)

# Data preparation
print("📊 Preparing training data...")

train_datagen = ImageDataGenerator(
rescale=1./255, # Normalize pixels to 0-1
validation_split=0.2, # Use 20% for validation
rotation_range=20, # Randomly rotate images
width_shift_range=0.2, # Randomly shift horizontally
height_shift_range=0.2, # Randomly shift vertically
horizontal_flip=True # Randomly flip images
)

# Training data generator
train_generator = train_datagen.flow_from_directory(
'images', # Our images folder
target_size=(150, 150), # Resize all images
batch_size=16, # Process 16 images at once
class_mode='categorical', # Multiple classes
subset='training' # Training set
)

# Validation data generator
val_generator = train_datagen.flow_from_directory(
'images',
target_size=(150, 150),
batch_size=16,
class_mode='categorical',
subset='validation' # Validation set
)

print(f"✅ Found {train_generator.samples} training images")
print(f"✅ Found {val_generator.samples} validation images")
print(f"📂 Classes: {list(train_generator.class_indices.keys())}")

# Build the model
print("\n🧠 Building neural network...")

model = tf.keras.models.Sequential([
# First convolutional block
tf.keras.layers.Conv2D(32, (3,3), activation='relu', input_shape=(150,150,3)),
tf.keras.layers.MaxPooling2D(2,2),

# Second convolutional block
tf.keras.layers.Conv2D(64, (3,3), activation='relu'),
tf.keras.layers.MaxPooling2D(2,2),

# Classification layers
tf.keras.layers.Flatten(),
tf.keras.layers.Dense(64, activation='relu'),
tf.keras.layers.Dense(train_generator.num_classes, activation='softmax')
])

# Compile the model
model.compile(
optimizer='adam',
loss='categorical_crossentropy',
metrics=['accuracy']
)

print("✅ Model architecture created!")
print("\n📊 Model Summary:")
model.summary()

# Train the model
print("\n🏋️ Starting training...")
print("This will take 5-10 minutes depending on your computer")

history = model.fit(
train_generator,
validation_data=val_generator,
epochs=5, # Number of training cycles
verbose=1 # Show progress
)

# Save the trained model
model.save("sajat_ai_model.h5")
print("\n💾 Model saved as 'sajat_ai_model.h5'")
print("🎉 Training completed successfully!")

# Plot training history
plt.figure(figsize=(12, 4))

plt.subplot(1, 2, 1)
plt.plot(history.history['accuracy'], label='Training Accuracy')
plt.plot(history.history['val_accuracy'], label='Validation Accuracy')
plt.title('Model Accuracy')
plt.xlabel('Epoch')
plt.ylabel('Accuracy')
plt.legend()

plt.subplot(1, 2, 2)
plt.plot(history.history['loss'], label='Training Loss')
plt.plot(history.history['val_loss'], label='Validation Loss')
plt.title('Model Loss')
plt.xlabel('Epoch')
plt.ylabel('Loss')
plt.legend()

plt.tight_layout()
plt.savefig('training_results.png')
plt.show()

print(f"\n🎯 Final Results:")
print(f"Training Accuracy: {history.history['accuracy'][-1]:.1%}")
print(f"Validation Accuracy: {history.history['val_accuracy'][-1]:.1%}")

📈 What These Numbers Mean:

📊 Accuracy

Percentage of correct predictions

  • 0.5000 = 50% (random guessing)
  • 0.7000 = 70% (good)
  • 0.8500 = 85% (excellent)

📉 Loss

How "wrong" the predictions are

  • Lower is better
  • Should decrease over epochs
  • Good target: < 0.5

import tensorflow as tf
import numpy as np
from tensorflow.keras.preprocessing import image
import os

print("🔮 Cat vs Dog AI Predictor")
print("=" * 40)

# Load the trained model
print("📚 Loading your trained AI model...")
try:
model = tf.keras.models.load_model("sajat_ai_model.h5")
print("✅ Model loaded successfully!")
except Exception as e:
print(f"❌ Error loading model: {e}")
print("Make sure you've run train_model.py first!")
exit()

# Get class names from the images folder
class_names = sorted(os.listdir("images"))
print(f"🏷️ Classes: {class_names}")

def predict_image(img_path):
"""
Predict if an image contains a cat or dog
Returns: predicted class and confidence
"""
try:
# Load and preprocess the image
img = image.load_img(img_path, target_size=(150, 150))

# Convert to array and normalize (same as training)
img_array = image.img_to_array(img) / 255.0

# Add batch dimension (model expects batches)
img_array = np.expand_dims(img_array, axis=0)

# Make prediction
prediction = model.predict(img_array, verbose=0)

# Get the predicted class
predicted_class_idx = np.argmax(prediction)
predicted_class = class_names[predicted_class_idx]
confidence = float(prediction[0][predicted_class_idx]) * 100

return predicted_class, confidence

except Exception as e:
return None, f"Error: {str(e)}"

# Interactive prediction loop
print("\n🎯 Ready to test your AI!")
print("Commands:")
print(" • Enter image filename (e.g., test_cat.jpg)")
print(" • Type 'quit' to exit")
print("-" * 40)

img_path = input("\n📸 Enter image filename: ").strip()

if img_path.lower() == 'quit':
print("👋 Thanks for testing your AI!")
break

if not os.path.exists(img_path):
print("❌ File not found. Please check the filename and try again.")
continue

# Make prediction
predicted_class, confidence = predict_image(img_path)

if predicted_class:
# Determine emoji
emoji = "🐱" if "cat" in predicted_class.lower() else "🐕"

print(f"🔮 Prediction: {emoji} {predicted_class.upper()}")
print(f"📊 Confidence: {confidence:.1f}%")

# Fun confidence interpretation
if confidence > 90:
print("💪 Very confident!")
elif confidence > 75:
print("👍 Pretty sure!")
elif confidence > 60:
print("🤔 Somewhat confident")
else:
print("😅 Not very sure...")
else:
print(f"❌ {confidence}") # Error message

print("-" * 40)

import tensorflow as tf
import numpy as np
from tensorflow.keras.preprocessing import image
import os
from datetime import datetime

class CatDogClassifier:
def __init__(self, model_path="sajat_ai_model.h5"):
"""Initialize the Cat vs Dog AI Classifier"""
print("🐱🐶 Cat vs Dog AI Classifier")
print("=" * 40)
print("🔮 Loading AI model...")

try:
self.model = tf.keras.models.load_model(model_path)
print("✅ Model loaded successfully!")
except Exception as e:
print(f"❌ Error loading model: {e}")
return

self.class_names = self._get_class_names()
self.prediction_history = []
print(f"🏷️ Classes detected: {self.class_names}")
print("🚀 Ready to classify images!")
print("=" * 40)

def _get_class_names(self):
"""Get class names from images folder"""
if os.path.exists("images"):
return sorted(os.listdir("images"))
else:
return ["cats", "dogs"] # Default

def predict_single(self, img_path):
"""Predict a single image"""
try:
# Load and preprocess
img = image.load_img(img_path, target_size=(150, 150))
img_array = image.img_to_array(img) / 255.0
img_array = np.expand_dims(img_array, axis=0)

# Predict
prediction = self.model.predict(img_array, verbose=0)
predicted_class_idx = np.argmax(prediction)
confidence = float(prediction[0][predicted_class_idx]) * 100
predicted_class = self.class_names[predicted_class_idx]

return predicted_class, confidence, prediction[0]

except Exception as e:
return None, 0, f"Error: {str(e)}"

def batch_predict(self, folder_path):
"""Predict multiple images in a folder"""
results = []
image_files = [f for f in os.listdir(folder_path)
if f.lower().endswith(('.png', '.jpg', '.jpeg'))]

print(f"\n🔍 Analyzing {len(image_files)} images...")
print("-" * 50)

for img_file in image_files:
img_path = os.path.join(folder_path, img_file)
predicted_class, confidence, _ = self.predict_single(img_path)

if predicted_class:
results.append({
'filename': img_file,
'prediction': predicted_class,
'confidence': confidence
})

# Print result with emojis
emoji = "🐱" if "cat" in predicted_class.lower() else "🐕"
print(f"{emoji} {img_file:<20} → {predicted_class} ({confidence:.1f}%)")

return results

def interactive_mode(self):
"""Interactive prediction mode"""
print("\n🎯 Interactive Mode Started!")
print("Commands:")
print(" • Enter image filename to predict")
print(" • 'batch [folder]' for batch prediction")
print(" • 'history' to see previous predictions")
print(" • 'quit' to exit")
print("-" * 50)

while True:
command = input("\n📸 Enter command or image filename: ").strip()

if command.lower() == 'quit':
print("👋 Thanks for using Cat vs Dog AI!")
break

elif command.lower() == 'history':
self._show_history()

elif command.lower().startswith('batch '):
folder = command[6:].strip()
if os.path.exists(folder):
self.batch_predict(folder)
else:
print(f"❌ Folder '{folder}' not found")

elif os.path.exists(command):
predicted_class, confidence, raw_pred = self.predict_single(command)

if predicted_class:
# Save to history
self.prediction_history.append({
'filename': command,
'prediction': predicted_class,
'confidence': confidence,
'timestamp': datetime.now().strftime("%H:%M:%S")
})

# Show detailed result
self._show_detailed_result(command, predicted_class, confidence, raw_pred)
else:
print(f"❌ Error processing {command}")

else:
print(f"❌ File '{command}' not found")

def _show_detailed_result(self, filename, predicted_class, confidence, raw_pred):
"""Show detailed prediction result"""
emoji = "🐱" if "cat" in predicted_class.lower() else "🐕"

print(f"\n📊 Detailed Analysis for '{filename}':")
print(f" {emoji} Prediction: {predicted_class.upper()}")
print(f" 📈 Confidence: {confidence:.1f}%")

# Show all class probabilities
print(f" 🔍 All probabilities:")
for i, class_name in enumerate(self.class_names):
prob = raw_pred[i] * 100
bar = "█" * int(prob / 5) # Simple bar chart
print(f" {class_name:<8}: {prob:5.1f}% {bar}")

# Confidence interpretation
if confidence > 90:
print(" 💪 Very confident prediction!")
elif confidence > 75:
print(" 👍 Confident prediction")
elif confidence > 60:
print(" 🤔 Moderate confidence")
else:
print(" 😅 Low confidence - image might be unclear")

def _show_history(self):
"""Show prediction history"""
if not self.prediction_history:
print("📜 No predictions yet!")
return

print(f"\n📜 Prediction History ({len(self.prediction_history)} predictions):")
print("-" * 70)
for pred in self.prediction_history[-10:]: # Show last 10
emoji = "🐱" if "cat" in pred['prediction'].lower() else "🐕"
print(f"{pred['timestamp']} | {emoji} {pred['filename']:<25} → {pred['prediction']} ({pred['confidence']:.1f}%)")

# Main execution
if __name__ == "__main__":
# Create classifier
classifier = CatDogClassifier()

# Check if model loaded successfully
if hasattr(classifier, 'model'):
# Start interactive mode
classifier.interactive_mode()
else:
print("🚫 Cannot start - model loading failed!")
