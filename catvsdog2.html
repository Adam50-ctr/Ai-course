import tensorflow as tf
from tensorflow.keras.preprocessing.image import ImageDataGenerator
import matplotlib.pyplot as plt
import os

print("ğŸ¤– Building Cat vs Dog AI Model")
print("=" * 40)

# Data preparation
print("ğŸ“Š Preparing training data...")

train_datagen = ImageDataGenerator(
rescale=1./255, # Normalize pixels to 0-1
validation_split=0.2, # Use 20% for validation
rotation_range=20, # Randomly rotate images
width_shift_range=0.2, # Randomly shift horizontally
height_shift_range=0.2, # Randomly shift vertically
horizontal_flip=True # Randomly flip images
)

# Training data generator
train_generator = train_datagen.flow_from_directory(
'images', # Our images folder
target_size=(150, 150), # Resize all images
batch_size=16, # Process 16 images at once
class_mode='categorical', # Multiple classes
subset='training' # Training set
)

# Validation data generator
val_generator = train_datagen.flow_from_directory(
'images',
target_size=(150, 150),
batch_size=16,
class_mode='categorical',
subset='validation' # Validation set
)

print(f"âœ… Found {train_generator.samples} training images")
print(f"âœ… Found {val_generator.samples} validation images")
print(f"ğŸ“‚ Classes: {list(train_generator.class_indices.keys())}")

# Build the model
print("\nğŸ§  Building neural network...")

model = tf.keras.models.Sequential([
# First convolutional block
tf.keras.layers.Conv2D(32, (3,3), activation='relu', input_shape=(150,150,3)),
tf.keras.layers.MaxPooling2D(2,2),

# Second convolutional block
tf.keras.layers.Conv2D(64, (3,3), activation='relu'),
tf.keras.layers.MaxPooling2D(2,2),

# Classification layers
tf.keras.layers.Flatten(),
tf.keras.layers.Dense(64, activation='relu'),
tf.keras.layers.Dense(train_generator.num_classes, activation='softmax')
])

# Compile the model
model.compile(
optimizer='adam',
loss='categorical_crossentropy',
metrics=['accuracy']
)

print("âœ… Model architecture created!")
print("\nğŸ“Š Model Summary:")
model.summary()

# Train the model
print("\nğŸ‹ï¸ Starting training...")
print("This will take 5-10 minutes depending on your computer")

history = model.fit(
train_generator,
validation_data=val_generator,
epochs=5, # Number of training cycles
verbose=1 # Show progress
)

# Save the trained model
model.save("sajat_ai_model.h5")
print("\nğŸ’¾ Model saved as 'sajat_ai_model.h5'")
print("ğŸ‰ Training completed successfully!")

# Plot training history
plt.figure(figsize=(12, 4))

plt.subplot(1, 2, 1)
plt.plot(history.history['accuracy'], label='Training Accuracy')
plt.plot(history.history['val_accuracy'], label='Validation Accuracy')
plt.title('Model Accuracy')
plt.xlabel('Epoch')
plt.ylabel('Accuracy')
plt.legend()

plt.subplot(1, 2, 2)
plt.plot(history.history['loss'], label='Training Loss')
plt.plot(history.history['val_loss'], label='Validation Loss')
plt.title('Model Loss')
plt.xlabel('Epoch')
plt.ylabel('Loss')
plt.legend()

plt.tight_layout()
plt.savefig('training_results.png')
plt.show()

print(f"\nğŸ¯ Final Results:")
print(f"Training Accuracy: {history.history['accuracy'][-1]:.1%}")
print(f"Validation Accuracy: {history.history['val_accuracy'][-1]:.1%}")

import tensorflow as tf
import numpy as np
from tensorflow.keras.preprocessing import image
import os
from datetime import datetime

class CatDogClassifier:
def __init__(self, model_path="sajat_ai_model.h5"):
"""Initialize the Cat vs Dog AI Classifier"""
print("ğŸ±ğŸ¶ Cat vs Dog AI Classifier")
print("=" * 40)
print("ğŸ”® Loading AI model...")

try:
self.model = tf.keras.models.load_model(model_path)
print("âœ… Model loaded successfully!")
except Exception as e:
print(f"âŒ Error loading model: {e}")
return

self.class_names = self._get_class_names()
self.prediction_history = []
print(f"ğŸ·ï¸ Classes detected: {self.class_names}")
print("ğŸš€ Ready to classify images!")
print("=" * 40)

def _get_class_names(self):
"""Get class names from images folder"""
if os.path.exists("images"):
return sorted(os.listdir("images"))
else:
return ["cats", "dogs"] # Default

def predict_single(self, img_path):
"""Predict a single image"""
try:
# Load and preprocess
img = image.load_img(img_path, target_size=(150, 150))
img_array = image.img_to_array(img) / 255.0
img_array = np.expand_dims(img_array, axis=0)

# Predict
prediction = self.model.predict(img_array, verbose=0)
predicted_class_idx = np.argmax(prediction)
confidence = float(prediction[0][predicted_class_idx]) * 100
predicted_class = self.class_names[predicted_class_idx]

return predicted_class, confidence, prediction[0]

except Exception as e:
return None, 0, f"Error: {str(e)}"

def batch_predict(self, folder_path):
"""Predict multiple images in a folder"""
results = []
image_files = [f for f in os.listdir(folder_path)
if f.lower().endswith(('.png', '.jpg', '.jpeg'))]

print(f"\nğŸ” Analyzing {len(image_files)} images...")
print("-" * 50)

for img_file in image_files:
img_path = os.path.join(folder_path, img_file)
predicted_class, confidence, _ = self.predict_single(img_path)

if predicted_class:
results.append({
'filename': img_file,
'prediction': predicted_class,
'confidence': confidence
})

# Print result with emojis
emoji = "ğŸ±" if "cat" in predicted_class.lower() else "ğŸ•"
print(f"{emoji} {img_file:<20} â†’ {predicted_class} ({confidence:.1f}%)")

return results

def interactive_mode(self):
"""Interactive prediction mode"""
print("\nğŸ¯ Interactive Mode Started!")
print("Commands:")
print(" â€¢ Enter image filename to predict")
print(" â€¢ 'batch [folder]' for batch prediction")
print(" â€¢ 'history' to see previous predictions")
print(" â€¢ 'quit' to exit")
print("-" * 50)

while True:
command = input("\nğŸ“¸ Enter command or image filename: ").strip()

if command.lower() == 'quit':
print("ğŸ‘‹ Thanks for using Cat vs Dog AI!")
break

elif command.lower() == 'history':
self._show_history()

elif command.lower().startswith('batch '):
folder = command[6:].strip()
if os.path.exists(folder):
self.batch_predict(folder)
else:
print(f"âŒ Folder '{folder}' not found")

elif os.path.exists(command):
predicted_class, confidence, raw_pred = self.predict_single(command)

if predicted_class:
# Save to history
self.prediction_history.append({
'filename': command,
'prediction': predicted_class,
'confidence': confidence,
'timestamp': datetime.now().strftime("%H:%M:%S")
})

# Show detailed result
self._show_detailed_result(command, predicted_class, confidence, raw_pred)
else:
print(f"âŒ Error processing {command}")

else:
print(f"âŒ File '{command}' not found")

def _show_detailed_result(self, filename, predicted_class, confidence, raw_pred):
"""Show detailed prediction result"""
emoji = "ğŸ±" if "cat" in predicted_class.lower() else "ğŸ•"

print(f"\nğŸ“Š Detailed Analysis for '{filename}':")
print(f" {emoji} Prediction: {predicted_class.upper()}")
print(f" ğŸ“ˆ Confidence: {confidence:.1f}%")

# Show all class probabilities
print(f" ğŸ” All probabilities:")
for i, class_name in enumerate(self.class_names):
prob = raw_pred[i] * 100
bar = "â–ˆ" * int(prob / 5) # Simple bar chart
print(f" {class_name:<8}: {prob:5.1f}% {bar}")

# Confidence interpretation
if confidence > 90:
print(" ğŸ’ª Very confident prediction!")
elif confidence > 75:
print(" ğŸ‘ Confident prediction")
elif confidence > 60:
print(" ğŸ¤” Moderate confidence")
else:
print(" ğŸ˜… Low confidence - image might be unclear")

def _show_history(self):
"""Show prediction history"""
if not self.prediction_history:
print("ğŸ“œ No predictions yet!")
return

print(f"\nğŸ“œ Prediction History ({len(self.prediction_history)} predictions):")
print("-" * 70)
for pred in self.prediction_history[-10:]: # Show last 10
emoji = "ğŸ±" if "cat" in pred['prediction'].lower() else "ğŸ•"
print(f"{pred['timestamp']} | {emoji} {pred['filename']:<25} â†’ {pred['prediction']} ({pred['confidence']:.1f}%)")

# Main execution
if __name__ == "__main__":
# Create classifier
classifier = CatDogClassifier()

# Check if model loaded successfully
if hasattr(classifier, 'model'):
# Start interactive mode
classifier.interactive_mode()
else:
print("ğŸš« Cannot start - model loading failed!")
